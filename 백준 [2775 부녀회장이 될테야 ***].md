# 백준 [2775 부녀회장이 될테야]
```python
#내 힘으로 풀고 싶었지만 아직 어려워서 다른 블로그 풀이를 봤다.
T = int(input())

for _ in range(T):  
    k = int(input())
    n = int(input())
    people = [x for x in range(1, n+1)]  #0층에서 1부터 n호까지의 각각 사람 수를 리스트에 담는다.

    for x in range(k):    #층 수 만큼 반복
        for y in range(1, n):   #1부터 끝까지
            people[y] += people[y-1] 
    print(people[-1])
```
