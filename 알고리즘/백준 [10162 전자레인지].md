# 백준 [10162 전자레인지]
```python
T = int(input())    #요리 시간(초)을 받는다.
A = T//300          #요리 시간을 5분으로 몇 번 나눌 수 있는지
T = T%300           #나머지로 T를 초기화
B = T//60           #요리 시간을 1분으로 몇 번 나눌 수 있는지
T = T%60            
C = T//10           #요리 시간을 10초로 몇 번 나눌 수 있는지
T = T%10
if T == 0:          #만약 요리 시간이 딱 떨어지면
    print(f"{A} {B} {C}")
else:               #요리 시간이 딱 떨어지지 않으면
    print(-1)
```
