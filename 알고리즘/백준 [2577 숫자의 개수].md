# 백준 [2577 숫자의 개수]
```python
N = 1                       #세 수의 곱을 나타니기 위한 초기값
for _ in range(3):          #세 수를 받아야하므로 3번 반복
    n = int(input())        #세 수를 하나씩 받는다.
    N *= n                  #N에 하나씩 곱해준다.
N = str(N)                  #정수를 문자열로 바꿔준다.(인덱싱을 이용하기 위해)
for i in range(10):         #0부터 9까지(총 10개) 몇 번 쓰였는지 확인하기 위해 반복
    print(N.count(str(i)))  #0부터 9까지 몇 번 쓰였는지 출력(찾으려는 값도 문자열로 해야한다.)
```
