# 백준 [11653 소인수분해]
```python
N = int(input())
i = 2       #항상 2부터 나누기 시작하니까
while N > 1:    #무한루프가 아니라 N이 1인 경우 아무것도 출력하지 않으니까 N이 1보다 크면 계속 반복하게 한다.
    if N % i == 0:  #N이 i로 나눠지면
        print(i)    #i(약수)를 출력
        N //= i     #i로 나눈 나머지를 N에 다시 넣는다.
    else:           #N이 i로 나눠지지 않으면
        i += 1      #i에 1을 더한다. 그리고 다시 반복
```
