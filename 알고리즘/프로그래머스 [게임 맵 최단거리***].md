# 2중 for문
# 1 발견?? -> DFS/BFS -> 0 으로 바꾸고 cnt +1
from collections import deque

T = int(input())

queue = deque([(0, 0)])

dr = [-1, 1, 0, 0]  # 상 하 좌 우
dc = [0, 0, -1, 1]

def BFS(cabbage):
    while queue:
        r, c = queue.popleft()

        if cabbage[r][c] == 1:
            cabbage[r][c] = 0
        
        for d in range(4):
            nr, nc = r + dr[d], c + dc[d]
            if 0 <= nr < N and 0 <= nc < M and cabbage[nr][nc] == 1:
                queue.append((nr, nc))


for _ in range(T):
    # M -> c / N -> r
    M, N, K = map(int, input().split())
    cabbage = [[0]*M for _ in range(N)]
    for _ in range(K):
        x, y = map(int, input().split())
        cabbage[y][x] = 1           # 여기까지 배추 심기 완료!
    
    cnt = 0

    for i in range(N):
        for l in range(M):
            if cabbage[i][l] == 1:
                BFS(cabbage)
                cnt += 1

    print(cnt)
